tamanhos=()
intervalos=(
    "1000 10000 500"
    "10000 100000 4000"
    "100000 1000000 30000"
    "1000000 10000000 200000"
    "10000000 100000001 1000000"
)
for intervalo in "${intervalos[@]}"; do
    read inicio fim passo <<< "$intervalo"
    for ((i=inicio; i<fim; i+=passo)); do
        tamanhos+=("$i")
    done
done
