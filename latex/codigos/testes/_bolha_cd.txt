void Bolha_CD(int tamanho, int *vetor, Dados *dados) {
    for (int i = tamanho - 1; i > 0; i--) {
        for (int j = 0; j < i; j++) {
            if (vetor[j] > vetor[j + 1]) {
                Troca(&vetor[j], &vetor[j + 1]);
                dados->movimentacoes += 3;
            }
            dados->comparacoes++;
        }
    }
}
