void Contagem(int n, int *v) {
    int *contagem = AlocaVetorLimpo(M + 1);
    int *v_aux = AlocaVetor(n);
    for (int i = 0; i < n; i++) contagem[v[i]]++;
    for (int i = 1; i <= M; i++) contagem[i] += contagem[i - 1];
    for (int i = n - 1; i >= 0; i--) {
        v_aux[contagem[v[i]] - 1] = v[i];
        contagem[v[i]]--;
    }
    CopiaVetor(n, v_aux, v);
    free(contagem);
    free(v_aux);
}
